class Player {

		UserName = NewName;
	}
	void Render() {}
private:
	string UserName;
};
int main()
{
	int mode;
	cout << "모드를 선택하세요(쓰기:1):";
	cin >> mode;
	if (mode == 1)
	{
		// 파일 쓰기
		ofstream file_op("D:\\file_read_test.txt");

		cout << "몇명 입력하시나요?";
		int HowMany;
		cin >> HowMany;
		string UserName;

		for(int i=0; i < HowMany; i++)
		{
			cin >> UserName;
			file_op << UserName << endl;
		}

		file_op.close();
	}
	else
	{
		vector<Player*> vcPlayers;

		// 파일 읽기 
		ifstream readFile;
		readFile.open("D:\\file_read_test.txt");

		if (readFile.is_open())
		{
			while (!readFile.eof())
			{
				string str;
				getline(readFile, str); //#include <string>


				cout << str << endl; //<== OK 이놈 아이디야

				Player* pl = new Player;
				pl->SetUserName(str);
				vcPlayers.push_back(pl);

			}
			readFile.close();
		}


		//=============================
		//엔진에서 플레이어 Render()
		for (Player* pl : vcPlayers)
		{
			pl->Render();
		}
	}
	
}

	virtual void Render() {}

};
class PlayerCharacter : public Character {
public:
	virtual void Update() override {}
	virtual void Render() override {
		cout << "PlayerCharacter Render()" << endl;
	}
};
class EnemyCharacter : public Character {

	virtual void Update() override {}
	virtual void Render() override {
		cout << "EnemyCharacter Render()" << endl;

	void AttackPlayer() {}
};

		Object* pl = new PlayerCharacter;
		vcObjects.push_back(pl);
	}
