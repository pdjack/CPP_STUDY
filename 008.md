#include "Map.h"
Map::Map()
{

}

void Map::ReadMap(string fileName)
{
	cout << fileName << endl;

    ifstream readFile;
    readFile.open(fileName, ios_base::in);

    if (readFile.is_open())
    {
        while (!readFile.eof())
        {
            string str;
            getline(readFile, str);//#include <string>
            
            cout << str << endl;
            //object = str;
            objectList.push_back(str);
        }
        readFile.close();
    }

}

void Map::Render()
{
    cout << "Render:" << endl;
    for (string str : objectList)
    {
        cout << str << endl;
    }
    
	
}
